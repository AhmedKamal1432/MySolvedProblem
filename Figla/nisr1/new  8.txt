int count(int r,int c){
	if(arr[r][c] == 'T')
		return 0;
	int count = 0;
	if(arr[r][c] != '.')
	   count++;
	   // up
	   int i = r-1;
	   while(i >= 0){
		   if(arr[i][c] == 'T' || arr[i][c] == 'T')
			   break;
		   if(arr[i][c] == 'V'){
			   count++;
			   break;
		   }
         i--;
	   }
	   //down
	       i = r+1;
	  	   while(i <n ){
	  		   if(arr[i][c] == 'T' ||arr[i][c] == 'H')
	  			   break;
	  		   if(arr[i][c] == 'V'){
	  			   count++;
	  			   break;
	  		   }
	           i++;
	  	   }

	  	  //right
	  	   int j = c-1;
	  	   while(j >= 0){
	  		   if(arr[r][j] == 'T'||arr[r][j] == 'V')
	  			   break;
	  		   if(arr[r][j] == 'H'){
	  			   count++;
	  			   break;
	  		   }
	           j--;
	  	   }

	  	   //left
	  	 j = c+1;
		 while(j <n ){
	  		  		   if(arr[r][j] == 'T' || arr[r][j] == 'V')
	  		  			   break;
	  		  		   if(arr[r][j] == 'H'){
	  		  			   count++;
	  		  			   break;
	  		  		   }
	  		           j++;
	  		  	   }
  return count;
}
